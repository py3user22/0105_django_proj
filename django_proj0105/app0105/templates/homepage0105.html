# git install notes 010524'
1- cd to directory want to have as repo
>> git init

2- check what branch currently on
>> git branch

3- add remote URL for repo to connect to on github
>> git remote add origin https://github.com/py3user22/0105_django_proj.git

4- check if remote is correct using
>> git remote -v
origin https://github.com/py3user22/0105_django_proj.git (fetch)
origin https://github.com/py3user22/0105_django_proj.git (push)

________# install pipenv
>> pip install pipenv

________# start shell
>> pipenv shell

_______# install django
>> pipenv install django

_______# startproject cmd
>> django-admin startproject django_proj0105

_______# startapp cmd -> change to directory with manage.py in it
>> py manage.py startapp app0105

# to save data in repo
>> git add .

# to commit to repo
>> git commit -m '0105 enter message here'

# to push to remote repo
>> git push -u origin main

# to pull from remote repo
>> git pull origin main

# optional: if have merge history error
>> git merge origin/main --allow-unrelated-histories
